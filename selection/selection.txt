Procedure SelectionSort (input/output s1, s2, ..., sn : integer)
{ Mengurutkan s1, s2, ..., sn secara menaik menggunakan selection sort }

Deklarasi
    i, j, imin, temp : integer

Algoritma
    for i ← 1 to n - 1 do
        imin ← i
        for j ← i + 1 to n do
            if s[j] < s[imin] then
                imin ← j
            endif
        endfor

        temp ← s[i]
        s[i] ← s[imin]
        s[imin] ← temp
    endfor
EndProcedure
